CREATE EXTENSION IF NOT EXISTS pgcrypto;   

CREATE TABLE products (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(50) NOT NULL, 
    sku VARCHAR(20) NOT NULL UNIQUE, 
    category VARCHAR(25) NOT NULL,
    brand VARCHAR(50) NOT NULL,
    unit VARCHAR(25) NOT NULL,
    sp DECIMAL(10,2) NOT NULL CONSTRAINT check_bp_sp CHECK(sp > bp),
    bp DECIMAL(10,2) NOT NULL,
    quantity INT, 
    minQty INT, 
    warehouse VARCHAR(50) NOT NULL
    );

CREATE TABLE IF NOT EXISTS users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(50) NOT NULL,
  role VARCHAR(20) NOT NULL,
  emp_date TIMESTAMPTZ,
  salary DECIMAL(10,2),
  password TEXT NOT NULL
  
);

CREATE TABLE IF NOT EXISTS suppliers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  address VARCHAR(50),
  email VARCHAR(50) NOT NULL,
  phone VARCHAR(20) NOT NULL,
  speciality VARCHAR(50),
  authorizer UUID REFERENCES users(id)
); 


CREATE TABLE IF NOT EXISTS expenses (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  e_date TIMESTAMPTZ DEFAULT now(),
  payment_method VARCHAR(25) NOT NULL,
  mpesa_code VARCHAR(25) CONSTRAINT require_mpesa CHECK (payment_method ='mpesa'),
  cheque_no VARCHAR(25) CONSTRAINT require_cheque CHECK (payment_method ='cheque'),
  items JSONB,
  total DECIMAL(10,2),
  notes TEXT, 
  supplier UUID REFERENCES suppliers(id),
  product UUID REFERENCES products(id),
  authorizer UUID REFERENCES users(id)
  
);

CREATE TABLE IF NOT EXISTS customers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  phone VARCHAR(20),
  email VARCHAR(50),
  no_of_orders INT CONSTRAINT no_of_orders_non_negative CHECK (no_of_orders >= 0)
  
);

CREATE TABLE IF NOT EXISTS invoices (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_id UUID REFERENCES customers(id),
  supplier_id UUID REFERENCES suppliers(id),
  cart JSONB, 
  total DECIMAL(10,2),
  i_date TIMESTAMPTZ,
);
